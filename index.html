#include #include #include #include #include #include // ========================================== // 1. CREDENCIALES DE THINGER.IO Y WIFI // ========================================== #define USER_ID "edson75" #define DEVICE_ID "Puente_ESP32" #define DEVICE_CREDENTIAL "TU_DEVICE_CREDENTIAL_AQUI" // La clave secreta de tu dispositivo en Thinger.io const char* SSID = "TU_RED_WIFI"; const char* PASSWORD = "TU_PASSWORD_WIFI"; ThingerESP32 thing(USER_ID, DEVICE_ID, DEVICE_CREDENTIAL); // ========================================== // 2. CONFIGURACIÓN DE SENSORES (PINES) // ========================================== // Sensor DHT22 (Temperatura/Humedad) #define DHTPIN 4 #define DHTTYPE DHT22 DHT dht(DHTPIN, DHTTYPE); // Sensor HC-SR04 (Ultrasonido - Nivel de agua) #define TRIG_PIN 5 #define ECHO_PIN 18 // Sensor MPU6050 (Inclinómetro / Acelerómetro) Adafruit_MPU6050 mpu; void setup() { Serial.begin(115200); // Conexión WiFi thing.add_wifi(SSID, PASSWORD); // Inicializar DHT22 dht.begin(); // Inicializar HC-SR04 pinMode(TRIG_PIN, OUTPUT); pinMode(ECHO_PIN, INPUT); // Inicializar MPU6050 Wire.begin(21, 22); // SDA=21, SCL=22 if (!mpu.begin()) { Serial.println("No se encontró el chip MPU6050"); } // ========================================== // 3. RECURSO "sensores" PARA TU PAGINA WEB // ========================================== thing["sensores"] >> [](pson& out){ // Lectura Ultrasonido (Distancia) digitalWrite(TRIG_PIN, LOW); delayMicroseconds(2); digitalWrite(TRIG_PIN, HIGH); delayMicroseconds(10); digitalWrite(TRIG_PIN, LOW); long duration = pulseIn(ECHO_PIN, HIGH); float distancia = duration * 0.034 / 2.0; // Lectura DHT22 (Temperatura) float temp = dht.readTemperature(); if (isnan(temp)) temp = 0.0; // Lectura MPU6050 (Inclinación) sensors_event_t a, g, temp_mpu; mpu.getEvent(&a, &g, &temp_mpu); float rotX = a.acceleration.x / 9.81; // En unidades G float rotY = a.acceleration.y / 9.81; // Asignación de variables de salida out["distancia_agua"] = distancia; out["temperatura"] = temp; out["inclinacion_x"] = rotX; out["inclinacion_y"] = rotY; }; } void loop() { thing.handle(); }
